# Loops (while->for)
  
**ZigC**
```
from std.testing import expect;

test "loop continue expression" {
    for (size_t i = 0; i < 10; i += 1) {}
    // try expect(i == 10); // i not visible
}

test "loop continue expression, more complicated" {
    for (size_t i = 1, j = 1; i * j < 2000; i *= 2, j *= 3) {
        const my_ij = i * j;
        try expect(my_ij < 2000);
    }
}
```

## loop as an expresstion
  
**ZigC**
```
from std.testing import expect;

test "loop else" {
    try expect(rangeHasNumber(0, 10, 5));
    try expect(!rangeHasNumber(0, 10, 15));
}

bool rangeHasNumber(size_t begin, size_t end, size_t number) {
    return for (size_t i = begin; i < end; i += 1) {
        if (i == number) {
            break true;
        }
    } else false;
}
```
  

**Zig**
```

## loop with optionals
  
**ZigC**
```
from std.testing import expect;

test "loop null capture" {
    u32 sum1 = 0;
    numbers_left = 3; // ??
    for (u32? value = eventuallyNullSequence()) {
        sum1 += value;
    }
    try expect(sum1 == 3);

    u32 sum2 = 0;
    numbers_left = 3; // ??
    for (u32? value = eventuallyNullSequence()) {
        sum2 += value;
    } else {
        try expect(sum2 == 3);
    }
}

u32 numbers_left = undefined;
u32? eventuallyNullSequence() {
    return if (numbers_left == 0) null else blk: {
        numbers_left -= 1;
        break :blk numbers_left;
    };
}
```

## inline loop
  
**ZigC**
```
from std.testing import expect;

test "inline loop" {
    size_t sum = 0;
    inline for (comptime i = 0; i < 3; i += 1) {
        const T = switch (i) {
            case 0: f32;
            case 1: i8;
            case 2: bool;
            default: unreachable;
        };
        sum += typeNameLength(T);
    }
    try expect(sum == 9);
}

size_t typeNameLength(comptime tgpe T) {
    return @typeName(T).len;
}

u32 numbers_left;

u32!anyerror eventuallyErrorSequence() {
    return if (numbers_left == 0) error.ReachedZero else blk: {
        numbers_left -= 1;
        break :blk numbers_left;
    };
}
```

**ZigC**
```
from std.testing import expect;

test "for basics" {
    const items = int[]{ 4, 5, 3, 4, 0 };
    int sum = 0;

    for value in items {
        if (value == 0) {
            continue;
        }
        sum += value;
    }
    try expect(sum == 16);

    for value in items[0..1] {
        sum += value;
    }
    try expect(sum == 20);

    int sum2 = 0;
    for i, _ in items {
        try expect(@TypeOf(i) == size_t); // size_t ?? alias usize
        sum2 += @intCast(int, i);
    }
    try expect(sum2 == 10);
}

test "for reference" {
    int[] items = { 3, 4, 2 };

    for *value in items {
        *value += 1;
    }

    try expect(items[0] == 4);
    try expect(items[1] == 5);
    try expect(items[2] == 3);
}

test "for else" {
    int?[] items = { 3, 4, null, 5 };

    int sum = 0;
    const result = for value in items {
        if (value != null) {
            sum += value; // ?? '?' needed
        }
    } else blk: {
        try expect(sum == 12);
        break :blk sum;
    };
    try expect(result == 12);
}

    try expect(count == 8);
}
